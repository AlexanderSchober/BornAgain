\def\defineBox#1#2#3#4#5{
  \newmdenv[
    usetwoside=false,
    skipabove=3pt minus 1pt plus 3pt,
    skipbelow=3pt minus 1pt plus 3pt,
    leftmargin=-4pt,
    rightmargin=-4pt,
    innerleftmargin=2pt,
    innerrightmargin=2pt,
    innertopmargin=4pt,
    innerbottommargin=4pt,
    backgroundcolor=#3,
    topline=false,
    bottomline=false,
    linecolor=#4,
    linewidth=2pt,
    ]{#2*}
  \newenvironment{#1}
    {\begin{#2*}\makebox[0pt][r]{\smash{#5}}\ignorespaces}
    {\end{#2*}\mdbreakon}
}
